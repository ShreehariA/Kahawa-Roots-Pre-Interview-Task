\vspace{-0.6em}
\begin{itemize}\setlength{\itemsep}{1pt}
    \item every shipment gets a unique ID when first recorded
    \item raw data (form/CSV) → \textbf{Cloud Storage} as immutable backup

\begin{frame}{What Does All This Actually Mean?}
\scriptsize
\begin{columns}[T]

\begin{column}{0.48\textwidth}

\textbf{\faCubes\ Architecture Concepts}
\vspace{0.2em}

\begin{itemize}\setlength{\itemsep}{2pt}
    \item \textbf{\textcolor{kahawaBrown}{Microservices}} — instead of one big app, we split into 3 small independent ones (admin, API, QR page). if one breaks, the others keep running.

    \item \textbf{\textcolor{kahawaBrown}{Cloud Functions}} — code that runs \textit{only} when new data arrives. no server to manage, no cost when idle.

    \item \textbf{\textcolor{kahawaBrown}{REST API}} — the ``messenger'' between the admin form and database. checks, cleans, and passes data along.

    \item \textbf{\textcolor{kahawaBrown}{PostgreSQL}} — a database like a smart spreadsheet. stores shipments in organised tables with relationships.

    \item \textbf{\textcolor{kahawaBrown}{Medallion Architecture}} — data flows through 3 layers: \textcolor{bronzeColor}{\textbf{Bronze}} (raw backup) → \textcolor{silverColor}{\textbf{Silver}} (cleaned) → \textcolor{goldColor}{\textbf{Gold}} (customer-facing). no bad data reaches the QR page.
\end{itemize}

\end{column}

\begin{column}{0.48\textwidth}

\textbf{\faLock\ Operations \& Security}
\vspace{0.2em}

\begin{itemize}\setlength{\itemsep}{2pt}
    \item \textbf{\textcolor{kahawaBrown}{IAM}} — controls \textit{who} can do \textit{what}. admins add shipments, public can only view QR pages.

    \item \textbf{\textcolor{kahawaBrown}{Terraform (IaC)}} — infra defined in code, not clicked manually. one command rebuilds the whole system.

    \item \textbf{\textcolor{kahawaBrown}{CI/CD}} — every code push is auto-tested and deployed. no manual uploads, no ``works on my laptop'' issues.

    \item \textbf{\textcolor{kahawaBrown}{Secret Manager}} — stores passwords \& API keys securely. no accidental credential leaks.
\end{itemize}

\vspace{0.3em}
\textbf{\faChartLine\ Future Plans}
\vspace{0.2em}

\begin{itemize}\setlength{\itemsep}{2pt}
    \item \textbf{\textcolor{kahawaBrown}{BigQuery}} — analytics when we have enough data. ``fastest shipping regions?'' ``transit time trends?''

    \item \textbf{\textcolor{kahawaBrown}{Dashboard}} — Tableau/Looker Studio for founders to see charts without touching the database.
\end{itemize}

\end{column}

\end{columns}
